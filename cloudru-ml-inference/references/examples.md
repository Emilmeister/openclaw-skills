# Cloud.ru ML Inference — Python Examples

## Setup (common for all examples)

```python
import os
from uuid import UUID

import httpx
from evoapp_http.auth import IAM_Auth
from http_client_retries import with_retries_failed_requests, with_retries_if
from inference_activity.clients import IAM_HttpClient, ModelRunUsersApiClient, options
from inference_activity.clients.inference.call_model_run_client import CallModelApi


def is_service_unavailable(response: httpx.Response) -> bool:
    return response.status_code in (502, 503, 504)


CP_CONSOLE_KEY_ID = os.environ["CP_CONSOLE_KEY_ID"]
CP_CONSOLE_SECRET = os.environ["CP_CONSOLE_SECRET"]
PROJECT_ID = UUID(os.environ["PROJECT_ID"])

iam_client = IAM_HttpClient("https://iam.api.cloud.ru/")

auth = IAM_Auth(
    key_id=CP_CONSOLE_KEY_ID,
    key_secret=CP_CONSOLE_SECRET,
    iam_client=iam_client,
)

bff_client = ModelRunUsersApiClient(
    "https://console.cloud.ru",
    options.with_auth(auth),
    options.with_autogenerated_request_id(),
    options.with_autogenerated_traceparent(),
    with_retries_failed_requests(),
    with_retries_if(is_service_unavailable),
)

call_client = CallModelApi(
    "modelrun.inference.cloud.ru",
    options.with_autogenerated_request_id(),
    options.with_autogenerated_traceparent(),
    with_retries_failed_requests(),
)

# For inference with auth enabled on the model run, add:
# call_client_with_auth = CallModelApi(
#     "modelrun.inference.cloud.ru",
#     options.with_auth(auth),
#     ...
# )
```

## 3. Create Model Run — vLLM from HuggingFace

```python
from inference_activity.models.inference.bff.v1.requests import (
    ModelRunCreateItem, ModelSourceHuggingFace, Repository, Scaling,
    KeepAliveDuration, ScalingRulesConcurrencyType, ConcurrencyType,
)
from inference_activity.enums import (
    FrameworkTypeEnum, ResourcesTypeEnum, ModelTaskType,
)

body = ModelRunCreateItem(
    name="llama2-7b-chat",
    framework_type=FrameworkTypeEnum.VLLM,
    resource_type=ResourcesTypeEnum.GPU_A100,
    model_task_type=ModelTaskType.TEXT_2_TEXT_GENERATION,
    model_source=ModelSourceHuggingFace(
        hugging_face_repository=Repository(
            repo="meta-llama/Llama-2-7b-chat-hf",
            revision="main",
        ),
    ),
    gpu_count=1,
    scaling=Scaling(
        min_scale=0,
        max_scale=1,
        scaling_rules=ScalingRulesConcurrencyType(
            concurrency_type=ConcurrencyType(soft=1, hard=2),
        ),
        keep_alive_duration=KeepAliveDuration(hours=0, minutes=15, seconds=0),
    ),
)

res = bff_client.v1.create_model_run(
    project_id=PROJECT_ID,
    model_run_body_data=body,
)
assert res.response.is_success
print(f"Created model run: {res.model().model_run_id}")
```

## 9. Call Model — OpenAI-compatible Chat (Ollama)

```python
from inference_activity.models.inference.call_model_run_client.v1.requests import (
    ModelTextToTextGenerationInputsRequestsOllama, ModelItemMessage,
)
from inference_activity.models.inference.call_model_run_client.v1.responses import (
    ModelTextToTextGenerationResponseOllama,
)

request_payload = ModelTextToTextGenerationInputsRequestsOllama(
    model="tinyllama:latest",
    messages=[ModelItemMessage(role="user", content="Why is the sky blue?")],
    keep_alive=1,
)

res = call_client.v1.call_model_run_with_compatible_openai(
    model_run_id=model_run_id,
    response_schema=ModelTextToTextGenerationResponseOllama,
    model_run_request_body_data=request_payload,
)
assert res.response.is_success
result = res.model()
print(result.choices[0].message.content)
print(f"Tokens used: {result.usage.total_tokens}")
```

## 17. Create Model Run with vLLM Dynamic Options and LoRA

```python
from inference_activity.models.inference.bff.v1.requests import (
    ModelRunCreateItem, ModelSourceHuggingFace, Repository,
    VllmOptionsDynamic, DynamicOptions, DynamicOption,
)
from inference_activity.enums import ParameterType

body = ModelRunCreateItem(
    name="llama2-with-lora",
    framework_type=FrameworkTypeEnum.VLLM,
    resource_type=ResourcesTypeEnum.GPU_A100,
    model_task_type=ModelTaskType.TEXT_2_TEXT_GENERATION,
    model_source=ModelSourceHuggingFace(
        hugging_face_repository=Repository(
            repo="meta-llama/Llama-2-7b-chat-hf",
        ),
    ),
    serving_options=VllmOptionsDynamic(
        dynamical_options=DynamicOptions(
            args=[
                DynamicOption(
                    key="--max-model-len",
                    value="4096",
                    parameter_type=ParameterType.NUMERIC,
                ),
                DynamicOption(
                    key="--enable-lora",
                    parameter_type=ParameterType.FLAG,
                ),
            ],
            lora_modules=[
                ModelSourceHuggingFace(
                    hugging_face_repository=Repository(
                        repo="my-org/my-lora-adapter",
                    ),
                ),
            ],
        ),
    ),
    gpu_count=1,
)
```
